class Stack(object):
  
  def __init__(self):
    # Initialize stack
    self.head = None # No head
    self.tail = None # No tail
    self.size = 0 # Size = 0
    pass 
  
  def push(self, val:int) -> None:
    """
    push operation
    """
    newhead =  Node(val) # New head value is val, newhead becomes a new node
    newhead.next = self.head # Self head is pushed back to be the next node after newhead
    self.head = newhead # The self.head need to be updated
    self.size += 1 # Add the size of self size
    return self
    pass

  def pop(self) -> int:
    """
    remove and return the value of top element in the stack
    """
    # Check if empty
    # If not, change reference, change size, return old top
    if self.size == 0: # Check the size of the node
      return None # Return nothing
    else:
      popnode = self.head # Pop the head of linked list
      self.head = self.head.next # Set up the self head equal to the next node 
      self.size -= 1 # Decrease the size of self
      return popnode.val # Return value that is popped
    pass 

  def peek(self) -> int:
    """
    return the value of top element
    """
    if self.size == 0: # Check if stack is empty before running
      return None # Return nothing
    else:
      peeknode = self.head # Set peek node as the top element 
      return peeknode.val # Return the value of peek node
    pass 

  def is_empty(self) -> bool:
    """
    return True if the stack is empty, otherwise return False
    """
    # If len = 0, return true, else return false
    if self.size == 0:
      return True 
    else:
      return False

#implement using this signature. Add anything else necessary.
class Queue(object):
  
  def __init__(self):
    self.head = None # Set head to none
    self.tail = None # Set tail to none
    self.size = 0 # Set the size of queue to 0
    pass 

  def enqueue(self, val:int) -> None:
    """
    insert val to the queue
    """
    enqueue_node = Node(val)  # Create a new node with the given value
    if self.tail:  # If the queue is not empty
      self.tail.next = enqueue_node  # Append the new node after the current tail
    else:  # If the queue is empty
      self.head = enqueue_node  # Set the new node as the head
    self.tail = enqueue_node  # Update the tail to the new node
    self.size += 1  # Increment the size
    pass 
  
  def dequeue(self) -> int :
    """
    remove and return the value from the queue
    """
    if self.size == 0: # If self is empty, return nothing
      return None
    else:
      dequeue_node = self.head  # Get the current head
      self.head = self.head.next  # Move the head to the next node
      if self.head is None:  # If the queue becomes empty after this operation
        self.tail = None  # Set the tail to None as well
      self.size -= 1  # decrease the size by 1
      return dequeue_node.val  # Return the value of the dequeued node
    pass 

  def is_empty(self):
    """
    return True if the queue is empty, otherwise return False
    """
   # If len = 0, return true, else return false
    if self.size == 0:
      return True 
    else: 
      return False
    pass

Problem 3: reverse a singly linked list (20 points)

Given the head of a singly linked list, reverse the list, and return the reversed list.

Example:

Questions: * what is the time and space complexity?

Time complexity of this method is \(O(n)\).

There are two operations before we entering the loop. (assigning value to prev and current)

The loops run for n iteration (Accessing the node n times, n operations), and for each round there are 4 operations inside the loop.

Therefore, there are 2 + n + 4 * n = 5n + 2 operations leading to time complexity of \(O(n)\).

For space complexity, although the linked list is stored, prev, current, and newnode stored a constant amount of data through iterations. Therefore, the space complexity is \(O(1)\).

#implement using this signature. Add anything else necessary.
def reverseList(self, head: Node) -> Node:
  prev = None
  # Set the current node to head
  current = head
  while current:
    # Set up a new node as the next element in a linked list
    newnode = current.next
    # Tell that the next from current is none
    current.next = prev
    # Move the node up one position
    prev = current
    # Move the current node up one position
    current = newnode
  return prev
  pass

Problem 4: Find the middle of a given singly linked list (20 points)

Given the head of a singly linked list, return the value of middle node of the linked list.

If there are two middle nodes, return the value of first middle node.

Example 1:

Questions: * what is the time and space complexity?

The time complexity of the function can be derived through dividing the processes.

Assigning value to current and length takes 2 operations.

For the first while loop, there are \(n\) nodes in the linked list.

For each node, there are 2 operations, += and assigning new value to currnet.

Therefore, in the first loop, there are 2 * n = 2n operations.

Between the loops, there are four assignments of final_length, middle, tally, and new_current.

For the second while loop, there are \(n/2\) loops.

In each round, there are two operations.

Therefore, the total number of operations in theis function is \(2 + 2n + 4 + 2*(\frac{n}{2}) = 3n + 6\) which has time complexity \(O(n)\)

For the space complexity, since length, tally, and middle are all integer variables, they have space complexity of \(O(1)\).

Even though the current and new_current are nodes in the linked list, they are just pointers and data of these nodes are not added into the data structure of this function.

Therefore, the space complexity of them is just \(O(1)\), and then totally the space complexity of whole function is \(O(1)\).

#implement using this signature. Add anything else necessary.
def middleNodeValue(self, head: Node) -> int:
    current = head
    length = 0
    while current:
        length += 1
        current = current.next
    final_length = length
    middle = (final_length-1)//2 
    tally = 0
    new_current = head
    while(tally != middle):
        tally += 1
        new_current = new_current.next
    return new_current.val

  • Use the Stack you implemented above.

  • On average the methods should have \(O(1)\) time complexity.

#implement using this signature. Add anything else necessary.
class TwoStacksQueue:

    def __init__(self):
      # Initialize the stack class
      self.stack1 = Stack()
      self.stack2 = Stack()
      pass
        

    def enqueue(self, x: int) -> None:
      "add an element at the end of a queue"
      self.stack1.push(x) # Add this to top of the stack is equivalent to the last of the queue
      return self
      pass
        

    def dequeue(self) -> int:
      "remove and return the front element of the queue"
      # Check if output stack is empty, if empty we have to transfer elements from output stack
      if self.stack2.is_empty():
        while not self.stack1.is_empty():
          self.stack2.push(self.stack1.pop()) # A method to reverse the linked list
        
      if not self.stack2.is_empty():
        return self.stack2.pop() # remove and return the first element of the reversed list = the head of queue
      else:
        return None
      pass 
        
    def is_empty(self) -> bool:
      # Check if self is empty or not
      if self.stack1.is_empty() and self.stack2.is_empty():
        return True
      else:
        return False
      pass

Bonus Problem: (stack) find number of waiting days to get a warmer day (20 points)

Given an array of integers temperatures represents the daily temperatures, return an array answer such that answer[i] is the number of days you have to wait after the ith day to get a warmer temperature. If there is no future day for which this is possible, keep answer[i] = 0 instead.

Solve the problem with time complexity O(n), where n is the length of input.

If you use stack, use Stack class you implemented above.

Example:

input = [30,28,29,50,40,45]
output = [3,1,1,0,1,0]

Exmplain: * On day 0 temperature=30, wait for 3 days to get a warmer day of temperatre 50 * On day 1 temperature=28, wait for 1 days to get a warmer day of temperatre 29 * On day 2 temperature=29, wait for 1 days to get a warmer day of temperatre 50 * On day 3 temperature=50, no warmer day in the future so set it to be 0 * On day 4 temperature=40, wait for 1 days to get a warmer day of temperatre 45 * On day 5 temperature=45, no warmer day in the future so set it to be 0

#implement using this signature. Add anything else necessary.
from typing import List
def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
  # Create a list of 0 of answer
  answer = [0] * len(temperatures)
  # Create a stack (For storing indexes)
  index = Stack()
  # Peek into a temperature list
  for today in range(len(temperatures)):
    # Check that the stack is not empty, and if not empty it will contains all the previous index before i.
    # We peek into the last thing we push in or basically first element of the stack
    while not index.is_empty() and temperatures[today] > temperatures[index.peek()]:
      # Get the previous day that have lower temp (index)
      previous_day = index.pop()  # Get the index of previous day, also at this point since we almost get the answer, remove this index from index stack too
      answer[previous_day] = today - previous_day # Updating answer, if we cannot find an index as a previousday of some day, that is it won't be updated and return 0
    index.push(today) # Put today at the end of list aka head of stack
  
  return answer # Return the updated answer
  pass
